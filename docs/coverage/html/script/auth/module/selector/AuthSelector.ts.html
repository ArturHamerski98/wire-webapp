export const REGISTER_FLOW = {
  GENERIC_INVITATION: 'REGISTER_FLOW_GENERIC_INVITATION',
  PERSONAL: 'REGISTER_FLOW_PERSONAL',
  TEAM: 'REGISTER_FLOW_TEAM',
};

  assets: [],
  email: undefined,

  label: undefined,
  locale: undefined,
  name: undefined,

  phone: undefined,

};
